258    def listen_to_update(self, callback) -> TaskRunningThread:
259        """
260        Listen to the server for updates.
261
262        This method connects to the server using a WebSocket connection.
263        The client will be updated in the following scenarios:
264        - Start or stop recording
265        - Lock or unlock the device
266        - Turn on or off eye encoding
267        - Change in mobile battery status
268        - When another client connects to the server to receive status updates
269
270        Arguments:
271        ----------
272            callback (Callable): The callback function to be called when an update is received.
273
274        Returns:
275        --------
276            TaskRunningThread: The thread object that is listening to updates.
277
278        Example:
279        --------
280        ```python
281        def callback(data):
282            print(data)
283
284        sc = SyncClient(address, port)
285        thread = sc.listen_to_update(callback)
286        thread.start()
287        ```
288        """
289        cancel_event = asyncio.Event()
290
291        async def _listen_task():
292            ws_uri = self._get_uri(Api.GET_STATUS.value, "ws")
293            listener = UpdateListener(ws_uri, callback)
294            await listener.begin_listen()
295            await cancel_event.wait()
296            await listener.end_listen()
297
298        return TaskRunningThread(_listen_task, cancel_event)

300    def create_streaming_thread(self, mode: StreamingMode) -> TaskRunningThread:
301        """
302        Creates a thread that runs the streaming task.
303
304        Arguments:
305        ----------
306            mode (StreamingMode): Specifies the streaming mode to be used,
307            enabling users to activate only the necessary streams.
308            If GAZE_SCENE is specified, both gaze and scene camera streams will be activated.
309
310        Returns:
311        --------
312            TaskRunningThread: The thread object that is running the streaming task.
313
314        Example:
315        --------
316        ```python
317        sc = SyncClient(address, port)
318        # Create a streaming thread that streams gaze and scene camera data.
319        th = sc.create_streaming_thread(StreamingMode.GAZE_SCENE)
320        th.start()
321
322        try:
323            while True:
324                frame_data = sc.get_scene_frames_from_streaming(timeout=5.0)
325                frame_datum = frame_data[-1] # get the last frame
326                buffer = frame_datum.get_buffer()
327                gazes = sc.get_gazes_from_streaming(timeout=5.0)
328                gaze = find_nearest_timestamp_match(frame_datum.get_timestamp(), gazes)
329
330                center = (int(gaze.combined.gaze_2d.x), int(gaze.combined.gaze_2d.y))
331                radius = 30
332                bgr_color = (255, 255, 0)
333                thickness = 5
334                cv2.circle(buffer, center, radius, bgr_color, thickness)
335
336                cv2.imshow('Press "q" to exit', buffer)
337                if cv2.waitKey(1) & 0xFF == ord('q'):
338                    break
339        except Exception as ex:
340            print(ex)
341        finally:
342            th.cancel()
343            th.join()
344            print('Stopped')
345        ```
346        """
347        cancel_event = asyncio.Event()
348           
349        async def _recv_task():
350            try:
351                if mode & StreamingMode.GAZE:
352                    gaze_url = self._rtsp_uri + Camera.GAZE.value
353                    gaze_stream = GazeStreamCollector(gaze_url, weakref.ref(self._data))
354                    gaze_stream.start()
355
356                if mode & StreamingMode.SCENE:
357                    scene_url = self._rtsp_uri + Camera.SCENE.value
358                    scene_stream = SceneCameraStreamCollector(scene_url, weakref.ref(self._data))
359                    scene_stream.start()
360
361                if mode & StreamingMode.EYES:
362                    left_eye_url = self._rtsp_uri + Camera.LEFT_EYE.value
363                    left_eye_stream = EyeCameraStreamCollector(left_eye_url, Camera.LEFT_EYE, weakref.ref(self._data))
364                    left_eye_stream.start()
365
366                    right_eye_url = self._rtsp_uri + Camera.RIGHT_EYE.value
367                    right_eye_stream = EyeCameraStreamCollector(right_eye_url, Camera.RIGHT_EYE, weakref.ref(self._data))
368                    right_eye_stream.start()
369
370                await cancel_event.wait()
371            except Exception as ex:
372                print(ex)
373        return TaskRunningThread(_recv_task, cancel_event)
