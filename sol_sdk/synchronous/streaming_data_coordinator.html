  4class StreamingDataCoordinator:
  5    """
  6    Class for coordinating streaming data from various sources.
  7
  8    This class manages the storage and retrieval of streaming data such as
  9    scene frames, gaze data, and left and right eye frames. It uses deques to
 10    store the data and threading events to signal the availability of new data.
 11    """
 12    def __init__(self, duration_seconds=3):
 13        """
 14        Initializes the instance with the specified duration.
 15        
 16        Arguments:
 17        ----------
 18        **duration_seconds** (int): The duration in seconds for which
 19            the data stores will retain frames. This value determines the
 20            maximum length of the deques based on the frame rate. For example,
 21            a duration of 3 seconds will store up to 90 frames for
 22            scene frames (30fps) and up to 360 frames for gaze and
 23            left eye frames (120fps).
 24        """
 25        # data stores
 26        self._scene_frame_store = deque(maxlen=duration_seconds*30) # 30fps
 27        self._gaze_store = deque(maxlen=duration_seconds*120) # 120fps
 28        self._left_eye_frame_store = deque(maxlen=duration_seconds*120) # 120fps
 29        self._right_eye_frame_store = deque(maxlen=duration_seconds*120) # 120fps
 30
 31        # events
 32        self._event_new_gaze_available = threading.Event()
 33        self._event_new_scene_data_available = threading.Event()
 34        self._event_new_left_eye_data_available = threading.Event()
 35        self._event_new_right_eye_data_available = threading.Event()
 36

 38    def add_gaze(self, gaze):
 39        """
 40        Adds a gaze data object to the gaze store.
 41        """
 42        self._gaze_store.append(gaze)
 43        self._event_new_gaze_available.set()
 44
 45    def pop_all_gazes(self):
 46        """
 47        Pops all gaze data objects from the gaze store.
 48
 49        Returns:
 50        ------
 51            list: The list of gaze data objects.
 52        """
 53        all_gazes = list(self._gaze_store)
 54        self._gaze_store.clear()
 55        return all_gazes
 56
 57    def wait_for_gazes(self, timeout=None):
 58        """
 59        Waits for new gaze data to be available.
 60
 61        Arguments:
 62        ----------
 63            timeout: The maximum time to wait for new gaze data.
 64
 65        Returns:
 66        -------
 67            list: The list of gaze data objects.
 68        """
 69        self._event_new_gaze_available.clear()
 70        if self._event_new_gaze_available.wait(timeout=timeout):
 71            return self.pop_all_gazes()
 72        return None

 76        Clears the gaze store.
