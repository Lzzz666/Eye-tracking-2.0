 14class TaskRunningThread(Thread):
 15    """
 16    A thread class that runs an asyncio task and supports cancellation.
 17    """
 18
 19    def __init__(self, task, cancel_event: asyncio.Event):
 20        """
 21        Initializes the thread with the task and the cancel event.
 22        """
 23        super().__init__()
 24        self._task = task
 25        self._cancel_event = cancel_event
 26
 27    def run(self):
 28        """
 29        Starts the asyncio event loop and runs the task.
 30
 31        This method is called when the thread is started. It initializes the asyncio
 32        event loop and runs the provided task.
 33        """
 34        asyncio.run(self._thread_func(
 35            obj_ref = weakref.ref(self),
 36            task = self._task
 37        ))
 38
 39    @staticmethod
 40    async def _thread_func(
 41        obj_ref: weakref.ReferenceType,
 42        task
 43    ):
 44        obj_ref()._loop = asyncio.get_event_loop()
 45        await task()
 46        obj_ref()._loop = None
 47
 48    def cancel(self):
 49        """
 50        Cancels the running task by setting the cancel event.
 51        """
 52        if self._loop:
 53            self._loop.call_soon_threadsafe(self._set_cancel_event)
 54
 55    def _set_cancel_event(self):
 56        self._cancel_event.set()

 60    A class that listens for updates from a WebSocket server.

 63        """
 64        Initializes the UpdateListener with the specified URI and
 65        callback function, which is called when an update is received.

 69        self._callback = callback
 70        """ The callback function to handle the received messages. """
 71
 72    async def begin_listen(self):
 73        """ Starts the listening task asynchronously. """
 74        self._update_task = asyncio.create_task(self.listen_func())
 75
 76    async def end_listen(self):
 77        """  Cancels the listening task if it is running and waits for it to finish."""
 78        if self._update_task is not None:
 79            self._update_task.cancel()
 80
 81            try:
 82                await self._update_task
 83            except asyncio.CancelledError:
 84                pass
 85            self._update_task = None
 86    
 87    async def listen_func(self):
 88        """
 89        The main listening function that connects to the WebSocket server
 90        and listens for updates.
 91        """
 92        async for msg in collect_update(self._uri):
 93            self._callback(msg)

 95async def collect_update(uri):
 96    """
 97    Asynchronously collect updates from a WebSocket server.
 98    """
 99    async for websocket in websockets.connect(uri):
100        try:
101            async for msg in websocket:
102                yield msg
103        except websockets.ConnectionClosed:
104            print("Connection closed. Retry connect...")
105            continue
106        except asyncio.CancelledError:
107            print("Task canceled")
108            break

124def find_nearest_timestamp_match(reference_timestamp, data_list: List[T]) -> T:
125    """

157def _validate_sdk_version_internal(sdk_version, api_version) -> bool:
158    '''
159    >>> _validate_sdk_version_internal('1.0.0', '1.0.0')
160    True
161    >>> _validate_sdk_version_internal('0.0.2', '1.0.0')
162    Traceback (most recent call last):
163    Exception: Your SDK version is outdated, please update to 1.0.0.
164    >>> _validate_sdk_version_internal('2.0.0', '1.0.0')
165    Traceback (most recent call last):
166    Exception: Your SDK version is newer, please update the Ganzin Chronus.
167    >>> _validate_sdk_version_internal('1.1.0', '1.0.0')
168    Warning: Your SDK version is newer in terms of minor versions, please update the Ganzin Chronus.
169    False
170    >>> _validate_sdk_version_internal('1.0.0', '1.1.0')
171    Warning: Your SDK version is outdated in terms of minor versions, please update to 1.1.0.
172    False
173    >>> _validate_sdk_version_internal('1.0.1', '1.0.0')
174    Note: Your SDK version is newer in terms of patch versions.
175    False
176    >>> _validate_sdk_version_internal('1.0.0', '1.0.1')
177    Note: Your SDK version is outdated in terms of patch versions, please update to 1.0.1.
178    False
179    '''
180
181    sdk_version = parse_semantic_version(sdk_version)
182    api_version = parse_semantic_version(api_version)
183
184    equal, is_sdk_newer, difference = compare_semantic_version(sdk_version, api_version)
185    if equal:
186        return True
187    else:
188        if is_sdk_newer:
189            msg1 = '{}Your SDK version is newer{}'
190            msg2 = ', please update the Ganzin Chronus.'
191            if difference == 'major':
192                raise Exception(msg1.format('', '') + msg2)
193            elif difference == 'minor':
194                print(msg1.format('Warning: ', ' in terms of minor versions') + msg2)
195                return False
196            elif difference == 'patch':
197                print(msg1.format('Note: ', ' in terms of patch versions') + '.')
198                return False
199        else:
200            msg = '{}Your SDK version is outdated{}, please update to {}.'
201            if difference == 'major':
202                raise Exception(msg.format('', '', api_version))
203            elif difference == 'minor':
204                print(msg.format('Warning: ', ' in terms of minor versions', api_version))
205                return False
206            elif difference == 'patch':
207                print(msg.format('Note: ', ' in terms of patch versions', api_version))
208                return False

210def parse_semantic_version(version):
211    """
212    Parse a semantic version string into a SemanticVersion object
213    which contains major, minor, and patch versions.
214    """
215    pattern = r"^(?P<major>\d+)\.(?P<minor>\d+)\.(?P<patch>\d+)$"
216    match = re.match(pattern, version)
217    if match:
218        return SemanticVersion(
219            major=int(match.group("major")),
220            minor=int(match.group("minor")),
221            patch=int(match.group("patch"))
222        )
223    else:
224        raise ValueError(f"Invalid semantic version: {version}")

226def compare_semantic_version(v1: SemanticVersion, v2: SemanticVersion):
227    '''
228    Compare two semantic versions.
229
230    Args:
231        v1 (SemanticVersion): The first semantic version to compare.
232        v2 (SemanticVersion): The second semantic version to compare.
233
234    Returns:
235        tuple: A tuple containing three elements:
236            - bool: True if the versions are equal, False otherwise.
237            - bool or None: If not equal, True if v1 is greater, False if v1 is lesser,
238              None if versions are equal.
239            - str or None: The part of the version that differs ('major', 'minor', 'patch'),
240              or None if versions are equal.
241    Raises:
242        ValueError: If the comparison result of the two arguments is unexpected.
243    '''
244    if v1 == v2:
245        return (True, None, None)
246    else:
247        if v1.major > v2.major:
248            return (False, True, 'major')
249        elif v1.major < v2.major:
250            return (False, False, 'major')
251        else:
252            if v1.minor > v2.minor:
253                return (False, True, 'minor')
254            elif v1.minor < v2.minor:
255                return (False, False, 'minor')
256            else:
257                if v1.patch > v2.patch:
258                    return (False, True, 'patch')
259                elif v1.patch < v2.patch:
260                    return (False, False, 'patch')
261                else:
262                    raise ValueError(f"Invalid version comparison: {v1}, {v2}")
