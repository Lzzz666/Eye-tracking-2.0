 55    def __init__(self, url):
 56        """
 57        Initializes the TimeSync with the specified URL.

 61    async def polls_multiple_times(self, number_of_polls = 60) -> TimeSyncResult:
 62        """
 63        Polls the server multiple times to get time offset and round trip time.
 64        """
 65        time_offset_results = list()
 66        round_trip_results = list()
 67
 68        async with websockets.connect(self._url) as ws:
 69            try:
 70                await self.send_sync_request(ws) # Ignore first result due to potentially expensive startup costs
 71
 72                for _ in range(number_of_polls):
 73                    try:
 74                        time_offset, round_trip = await self.send_sync_request(ws)
 75                        time_offset_results.append(time_offset)
 76                        round_trip_results.append(round_trip)
 77                    except ValueError as err:
 78                        print(err)
 79            except Exception as ex:
 80                print(ex)
 81
 82            return TimeSyncResult(
 83                TimeSyncStatistics(time_offset_results),
 84                TimeSyncStatistics(round_trip_results)
 85            )
 86
 87    @staticmethod
 88    async def send_sync_request(websocket: WebSocketClientProtocol):
 89        """
 90        Sends a time synchronization request to the server and returns the time offset and round trip time.
 91        """
 92        client_start_time = get_timestamp_ms()
 93        request = pack("!Q", client_start_time)
 94        await websocket.send(request)
 95        resp = await websocket.recv()
 96        client_end_time = get_timestamp_ms()
 97
 98        if len(resp) != 16:
 99            raise ValueError(f'Unexpected response length: {len(resp)}')
100
101        start_time_from_resp, server_time = unpack("!QQ", resp)
102        if start_time_from_resp != client_start_time:
103            raise ValueError(f'Echoed time mismatched')
104
105        projected_server_time = (client_start_time + client_end_time) / 2
106        time_offset = projected_server_time - server_time
107        round_trip = client_end_time - client_start_time
108        return (round(time_offset), round_trip)
