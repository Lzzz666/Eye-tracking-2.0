 22    """
 23    Asynchronous client for interacting with the server.
 24    """
 25    def __init__(self, address, port, rtsp_port=8086):
 26        """
 27        Initializes the AsyncClient instance with the specified address and ports.
 28
 29        Arguments:
 30        ----------
 31        - **address** : str
 32            The IP address of the server.
 33        - **port** : int
 34            The port number of the server.
 35        - **rtsp_port** : int, optional
 36            The RTSP port number of the server (default is 8086).
 37        """
 38        self._uri = f"{address}:{port}"
 39        self._rtsp_uri = f"{address}:{rtsp_port}"
 40        self._session = aiohttp.ClientSession()
 41
 42    async def _close(self):
 43        """
 44        Closes the client session.
 45
 46        This method releases resources, particularly the HTTP client session,
 47        ensuring that all connections are properly closed and cleaned up.
 48        """
 49        await self._session.close()
 50
 51    async def __aenter__(self) -> "AsyncClient":
 52        """
 53        Asynchronous context manager entry method.
 54
 55        Retrieves the remote API version and validates it against the SDK version.
 56        Ensures compatibility before returning the AsyncClient instance.
 57        If the major version differs, raises an exception.
 58
 59        Returns:
 60            AsyncClient: The current instance of the AsyncClient.
 61        """
 62        remote_api_version = (await self.get_version()).remote_api_version
 63        validate_sdk_version(remote_api_version)
 64        return self
 65
 66    async def __aexit__(self, exc_type, exc_value, traceback) -> "None":
 67        """
 68        Asynchronous context manager exit method.
 69        """
 70        await self._close()

 72    def _get_uri(self, api: Api, protocol="http"):
 73        return f"{protocol}://{self._uri}/api/{api}"

 76        """
 77        Retrieves the SDK version that the server is expected to support.

 79        Returns:

 92            return json.loads(await resp.text(), object_hook=versionResponseMapper)

330async def recv_gaze(ac: AsyncClient, *args, **kwargs):
331    """
332    Asynchronously receives gaze data from a gaze stream.
333
334    This function connects to a gaze stream using the provided AsyncClient instance,
335    and yields gaze data as it is received from the stream.
336
337    Arguments:
338    ----------
339        - ac (AsyncClient): An instance of AsyncClient to connect to the gaze stream.
340        - *args: Variable length argument list to pass to the gaze stream.
341        - **kwargs: Arbitrary keyword arguments to pass to the gaze stream.
342
343    Yields:
344    -------
345        GazeData: Parsed gaze data received from the stream.
346
347    Example:
348    ```python
349    async for gaze_data in recv_gaze(ac):
350        process_gaze(gaze_data)
351    ```
352    """
353    async with ac._gaze_stream(*args, **kwargs) as stream:
354        async for data in stream.get_data():
355            yield data

357async def recv_video(ac: AsyncClient, target: Camera, *args, **kwargs):
358    """
359    Asynchronously receives video frames from a video stream.
360
361    This function connects to a video stream using the provided AsyncClient instance
362    and the specified target camera, and yields video frames as they are received from the stream.
363
364    Arguments:
365    ----------
366        - ac (AsyncClient): An instance of AsyncClient to connect to the video stream.
367        - target (Camera): The target camera to receive video from. For example, Camera.SCENE or Camera.LEFT_EYE.
368        - *args: Variable length argument list to pass to the video stream.
369        - **kwargs: Arbitrary keyword arguments to pass to the video stream.
370
371    Yields:
372    -------
373        FrameData: Parsed video frame data received from the stream.
374
375    Example:
376    ```python
377    async for frame in recv_video(ac, target):  
378        process_frame(frame)
379    ```
380    """
381    async with ac._video_stream(target, *args, **kwargs) as stream:
382        async for frame in stream.get_data():
383            yield frame
