 40    def handle_raw_data(self, data: bytes, timestamp):
 41        """
 42        @private
 43        Processes raw data and decodes it into frame data.
 44
 45        This function handles raw byte data, initializes the codec if it is not already set,
 46        and decodes the data into frames. If the codec is not initialized, it creates the codec
 47        and parses the necessary parameters. Once the codec is ready, it processes the raw data
 48        to extract frames and associates them with the provided timestamp.
 49
 50        Args:
 51            data (bytes): The raw byte data to be processed.
 52            timestamp: The timestamp associated with the raw data.
 53
 54        Returns:
 55            FrameData: The decoded frame data along with the timestamp, or None if an error occurs
 56            during codec initialization.
 57        """
 58        if not self._codec:
 59            try:
 60                self._codec = self._create_codec()
 61                for param in self._get_sprop_parameters():
 62                    self._codec.parse(param)
 63            except (KeyError, IndexError) as error:
 64                return None
 65        else:
 66            for pkt in self._codec.parse(self._get_nal_payload(data)):
 67                for frame in self._codec.decode(pkt):
 68                    return FrameData(frame, self._timestamp)
 69            self._timestamp = timestamp
 70
 71    def _create_codec(self):
 72        return av.CodecContext.create(self._get_encoding(), "r")
 73
 74    def _get_encoding(self):
 75        if self._encoding is None:
 76            self._encoding = self.session.sdp["medias"][0]["attributes"]["rtpmap"]["encoding"].lower()
 77        return self._encoding
 78
 79    def _get_sprop_parameters(self):
 80        if self._sprop_parameters is None:
 81            parameter_sets = self.session.sdp["medias"][0]["attributes"]["fmtp"]["sprop-parameter-sets"]
 82            parts = [part for part in parameter_sets.split(",")]
 83            param_sets = [base64.b64decode(part) for part in parts]
 84            self._sprop_parameters = [self._get_nal_payload(param) for param in param_sets]
 85        return self._sprop_parameters
 86
 87    def _get_nal_payload(self, data):
 88        start_bytes = b'\x00\x00\x00\x01'
 89
 90        payload_header = unpack('!B', data[:1])[0]
 91        is_forbidden_bit_one = payload_header & 0b10000000
 92        if is_forbidden_bit_one:
 93            raise Exception('Forbidden_zero_bit not zero')
 94        
 95        payload_offset = 0
 96        nal_unit_type = payload_header & 0b00011111
 97        if nal_unit_type == 28:
 98            payload_offset = 2
 99            fu_header = unpack('!B', data[1:2])[0]
100            if fu_header & 0b10000000:
101                p1 = payload_header & 0b11000000
102                p2 = fu_header & 0b00011111
103                return start_bytes + bytes((p1 + p2, )) + data[payload_offset:]
104            else:
105                return data[payload_offset:]
106                
107        return start_bytes + data[payload_offset:]
