236    def from_raw(raw: RawGazeData, timestamp):
237        """
238        Create a GazeData object from raw gaze data.
239        """
240        return GazeData(
241            GazeData.Combined(
242                GazeData.Combined.Gaze2D(raw.combinedSceneCamGazeValidity, raw.combinedSceneCamGazePositionX, raw.combinedSceneCamGazePositionY),
243                GazeData.Combined.Gaze3D(raw.combinedGazeValidity, raw.combinedGazePositionX, raw.combinedGazePositionY, raw.combinedGazePositionZ)
244            ),
245            GazeData.Eye(
246                raw.leftEyeConfidence,
247                EyeStatus(raw.leftEyeStatus),
248                GazeData.Eye.Gaze(
249                    raw.leftEyeGazeValidity,
250                    GazeData.Eye.Gaze.Origin(raw.leftEyeGazeOriginX, raw.leftEyeGazeOriginY, raw.leftEyeGazeOriginZ),
251                    GazeData.Eye.Gaze.Direction(raw.leftEyeGazeDirectionX, raw.leftEyeGazeDirectionY, raw.leftEyeGazeDirectionZ)
252                ),
253                GazeData.Eye.Pupil3D(raw.leftEyePupil3dValidity, raw.leftEyePupil3dDiameter),
254                GazeData.Eye.Pupil2D(raw.leftEyePupil2dPositionX, raw.leftEyePupil2dPositionY)
255            ),
256            GazeData.Eye(
257                raw.rightEyeConfidence,
258                EyeStatus(raw.rightEyeStatus),
259                GazeData.Eye.Gaze(
260                    raw.rightEyeGazeValidity,
261                    GazeData.Eye.Gaze.Origin(raw.rightEyeGazeOriginX, raw.rightEyeGazeOriginY, raw.rightEyeGazeOriginZ),
262                    GazeData.Eye.Gaze.Direction(raw.rightEyeGazeDirectionX, raw.rightEyeGazeDirectionY, raw.rightEyeGazeDirectionZ)
263                ),
264                GazeData.Eye.Pupil3D(raw.rightEyePupil3dValidity, raw.rightEyePupil3dDiameter),
265                GazeData.Eye.Pupil2D(raw.rightEyePupil2dPositionX, raw.rightEyePupil2dPositionY)
266            ),
267            timestamp = timestamp
268        )

279    Class for handling gaze data from RTSP streams.

281    def handle_raw_data(self, data: bytes, timestamp):
282        """
283        @private
284        Processes raw data and decodes it into gaze data.
285        
286        The function attempts to unpack the raw byte data using a predefined format string.
287        If successful, it creates a RawGazeData object from the unpacked data and converts it
288        into a GazeData object using the provided timestamp. If any error occurs during the
289        unpacking or processing, it prints an error message and re-raises the exception.
290
291        Arguments:
292        ----------
293            - data (bytes): The raw byte data to be processed.
294            - timestamp: The timestamp associated with the raw data.
295
296        Returns:
297        ----------
298            GazeData: The decoded gaze data.
299
300        Raises:
301        ----------
302            Exception: If the data cannot be unpacked or processed.
303        """
304        try:
305            unpacked_data = struct.unpack('!bddbddddbddddddbbddddbddddddbbddd', data)
306            raw = RawGazeData(*unpacked_data)
307            return GazeData.from_raw(raw, timestamp)
308        except Exception as e:
309            print(f'Failed to unpack gaze data: {e}.')
310            raise e
