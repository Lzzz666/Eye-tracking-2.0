  9# Dynamically create the _mapping dictionary by iterating over the provided classes.
 10# For each class, extract the field names using the dataclasses.fields function.
 11# Create a frozenset of these field names and use it as the key in the _mapping dictionary.
 12# The value for each key is the corresponding class itself.
 13# This is necessary when dealing with nested dataclasses.
 14def _create_mapping(*classes):
 15    mapping = {}
 16    for cls in classes:
 17        field_names = frozenset(field.name for field in fields(cls))
 18        mapping[field_names] = cls
 19    return mapping
 20
 21# Creates a class instance based on the provided dictionary and the mapping.
 22def _class_mapper(d: dict, mapping):
 23    for keys, cls in mapping.items():
 24        if keys.issubset(d.keys()):  # Check if the dictionary keys contain all required keys.
 25            filtered_dict = {k: v for k, v in d.items() if k in keys}  # Filter out extra keys.
 26            return cls(**filtered_dict)
 27    raise KeyError(f"No mapping found for keys: {frozenset(d.keys())}")

 29def parse_json_to_object(json_data: str, cls: type) -> object:
 30    """
 31    Parses a JSON string and converts it into an instance of the specified class.
 32    
 33    Arguments:
 34    ----------
 35        - json_data (str): A JSON string representing the data.
 36        - cls (type): The class type to which the JSON data should be converted.
 37        
 38    Returns:
 39    --------
 40        object: An instance of the specified class populated with the JSON data.
 41
 42    Example:
 43    --------
 44    ```python
 45    obj = parse_json_to_object(json_text, TagResponse)
 46    ```
 47    """
 48    data = json.loads(json_data)
 49
 50    # Get the set of valid field names for the class
 51    valid_fields = {f.name for f in fields(cls)}
 52
 53    # Filter out any unexpected fields
 54    filtered_data = {k: v for k, v in data.items() if k in valid_fields}
 55
 56    return cls(**filtered_data)

357    def from_raw(cls, source: CaptureRawResult):
358        """ Creates a CaptureResult instance from the provided CaptureRawResult object. """
359        image = base64.b64decode(source.sceneImgBase64)
360        gaze = GazeData.from_raw(source.gazeData, source.timestamp)
